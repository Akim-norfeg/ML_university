# Функция и её градиент
def f(x):
    return 3 * x**2

def df(x):
    return 6 * x

x0 = 5  #начальная точка
learning_rate = 0.1
tolerance = 1e-8
history = [(x0, f(x0))]

Градиентный спуск¶

In [5]:
while True:
    x1 = x0 - learning_rate * df(x0)
    history.append((x1, f(x1)))
    
    if np.abs(x1 - x0) < tolerance:
        break
    
    x0 = x1

def f2(x, y):
    return 3*x**2 + 5*y**2

def df2(x, y):
    return np.array([6*x, 10*y])

x0 = np.array([5, 10])  #начальная точка
learning_rate = 0.1
tolerance = 1e-8
history = [(*x0, f2(*x0))]

Градиентный спуск¶

In [12]:
while True:
    x1 = x0 - learning_rate * df2(*x0)
    history.append((*x1, f2(*x1)))
    
    if np.linalg.norm(x1 - x0) < tolerance:
        break
    
    x0 = x1
